<script>
const slides = document.querySelectorAll(".slide");

let slideIndex = 0;
let transitionIndex = 0;
let autoTimer;

/* VALID TRANSITIONS */
const transitions = [
  "fade","push","wipe","split","reveal","bars","shape",
  "cover","uncover","flash","curtains","ripple",
  "honeycomb","cube","zoom","flip","pan","rotate","fly"
];

/* ===== MAIN SHOW FUNCTION ===== */
function showSlide(newIndex){

    const current = slides[slideIndex];

    slideIndex = newIndex;
    if(slideIndex >= slides.length) slideIndex = 0;
    if(slideIndex < 0) slideIndex = slides.length - 1;

    const next = slides[slideIndex];

    slides.forEach(slide=>{
        slide.className = "slide";
        slide.style.zIndex = 0;
    });

    const effect = transitions[transitionIndex];
    transitionIndex = (transitionIndex + 1) % transitions.length;

    next.classList.add(effect);
    next.classList.add("active");
    next.style.zIndex = 2;
}

/* ===== BUTTON CONTROLS ===== */
function nextSlide(){
    showSlide(slideIndex + 1);
    resetAuto();
}

function prevSlide(){
    showSlide(slideIndex - 1);
    resetAuto();
}

/* ===== AUTO SLIDE (4 seconds) ===== */
function startAuto(){
    autoTimer = setInterval(()=>{
        showSlide(slideIndex + 1);
    },4000);
}

function resetAuto(){
    clearInterval(autoTimer);
    startAuto();
}

/* START */
slides[0].classList.add("active");
startAuto();
</script>
